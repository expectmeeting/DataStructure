# 标准输入

## 1、C下的标准输入

~~~c
#include<stdio.h>
int scanf(const char *format, ...); //函数从标准输入流 stdin 读取输入，并根据提供的 format 来浏览输入。
//输入字符串
	char str[10];
    scanf("%s",str);//输入遇到空格或回车，scanf()就停止读取
	scanf("%[^\n]",str);//输入遇到回车，scanf()才停止读取
//但以上两种方法，都没有考虑溢出的问题，即输入字符数量超出存储大小，从而造成内存泄漏。
char *fgets(char *str, int n, FILE *stream);//从指定的流stream读取一行，并把它存储在str所指向的字符串内。当读取 (n-1) 个字符时，或者读取到换行符时，或者到达文件末尾时停止。
//例如：n = 5，输入"12345",则str指向的内存存储"1234"+'\0';输入"1234",则str指向的内存存储"1234"+'\0';输入"123",则str指向的内存存储"123"+'\n'+'\0';
//总结如下：若输入字符串长度为s,若 s < n-1,则存储输入字符串+"\n\0"；若 s >= n-1,则存储输入字符串前n-1个字符+"\0"；
    
~~~
